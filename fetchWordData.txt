                          const fTag = dmJson[0].tags.find(t => t.startsWith('f:'));
                          if (fTag) frequencyRaw = parseFloat(fTag.split(':')[1]);
                      }

              else if (frequencyRaw > 1) freqScore = 2;
              else freqScore = 1;
